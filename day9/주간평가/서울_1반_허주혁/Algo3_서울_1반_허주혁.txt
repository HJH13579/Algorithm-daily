def bubble_sort(a, N):

    # N = len(a)

    for i in range(N - 1, 0, -1):   # 시작은 전체적으로 돌리면서 마지막에 가장 큰 숫자 놓고 카운트 범위 -1씩
        for j in range(i):
            if a[j] > a[j + 1]: # 만약 앞이 뒤보다 더 크면
                a[j], a[j + 1] = a[j + 1], a[j] # 위치 교환

    return a
